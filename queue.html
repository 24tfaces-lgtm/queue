Настройка источника и колонок

1) Опубликуйте нужный лист Google Sheets в формате CSV и вставьте URL в CONFIG.csvUrl.
2) Если названия столбцов отличаются, поменяйте массивы в CONFIG.columns.
3) Этот шаблон читает данные только на просмотр. Для записи из дашборда нужен Apps Script / API. 
const CONFIG = {
  csvUrl: 'PASTE_PUBLISHED_CSV_URL_HERE',
  refreshMs: 60_000,
  columns: {
    orderId: ['ID', 'Номер заказа', 'Заказ'],
    operator: ['Оператор', 'Менеджер', 'Ответственный'],
    client: ['Клиент', 'Покупатель', 'Получатель'],
    createdAt: ['Дата заказа', 'Дата создания', 'Created At'],
    deliveryDate: ['Дата доставки', 'Плановая доставка', 'План'],
    deliveredAt: ['Факт доставки', 'Delivered At'],
    status: ['Статус', 'Статус доставки'],
    saleCheck: ['Чек продажи', 'Сумма продажи', 'Продажа'],
    buyoutAmount: ['Выкуп', 'Сумма выкупа', 'Оплачено'],
    buyoutFlag: ['Выкуплено', 'Статус выкупа', 'Buyout']
  }
};
